*** MODELINE
I don't really want to to dig into customization here, so I just use a package. Also, this really shouldn't be going in the aesthetics file, but it's fine
#+begin_src emacs-lisp
  (use-package punch-line
    :ensure t
    :defer t
    :hook ((after-init . punch-line-mode)        ;; Load punch-line
		   (after-init . punch-weather-update)   ;; Load weather
		   (after-init . punch-load-tasks))      ;; Load saved current tasks
    :config
    ;; Divider style
    (setq
     punch-line-left-separator "  "
     punch-line-right-separator "  "
     punch-line-vc-use-github-icon nil
     punch-show-weather-info nil
     punch-line-modal-size 'medium
     punch-line-music-info '(:service spotify)      ;; Music service configuration
     punch-line-music-max-length 80   ;; Max length of artist and song
     punch-line-modal-divider-style 'block)  ; Options: 'arrow, 'flame, 'ice, 'circle, 'block
    )
#+end_src
